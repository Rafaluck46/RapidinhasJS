/*
** O JavaScript possui formas diferentes de iterar objetos
** e arrays com um loop (for), na dica de hoje iremos ver 
** como funciona o 'for...of' para iterar um array. Veja como:

let array = [1, 3, 5, 7]

for (let valor of array) {
  console.log(valor);
}
